String.prototype.capitalize = function() {
      const phrase = this.toLowerCase();
      const isEmpty = !phrase.length;
      if(isEmpty) return phrase.valueOf();
      return phrase
        .split(' ')
        .map(word => 
          !word.length
            ? word
            : `${word[0].toUpperCase()}${word.slice(1, word.length)}`
        ).join(' ');
    };
